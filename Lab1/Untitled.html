import  matplotlib.pyplot as plt
import numpy as np

f = lambda x: x**2 + 5

def draw(x1, x2, f):
    x = np.linspace(x1, x2, 100)
    plt.plot(x, f(x))
    plt.title(f'Wykres funkcji x^2 +5 od {x1} do {x2}')
    plt.xlabel('Argument x')
    plt.ylabel('x^2 + 5')
    plt.legend(['Legenda'])
    plt.show()

x1 = np.linspace(-1, 1, 100)
x2 = np.linspace(-6, 6, 100)
x3 = np.linspace(0, 5, 100)
